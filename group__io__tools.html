!======================================================================
subroutine userchk
include 'TSTEP'
if (istep.eq.0) then
! start framework
call frame_start
endif
! monitor simulation
call frame_monitor
! finalise framework
if (istep.eq.nsteps.or.lastep.eq.1) then
call frame_end
endif
return
end
!======================================================================
subroutine frame_usr_register
implicit none
include 'SIZE'
include 'FRAMELP'
!-----------------------------------------------------------------------
! register modules
call io_register
return
end subroutine
!======================================================================

include 'FRAMELP'
!-----------------------------------------------------------------------
! initialise modules
return
end subroutine
!======================================================================

include 'SIZE'
include 'FRAMELP'
!-----------------------------------------------------------------------
! finlise modules
return
end subroutine
