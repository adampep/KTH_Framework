356  subroutine trip_1dprj()
357  implicit none
358 
359  include 'SIZE'
360  include 'INPUT'
361  include 'GEOM'
362  include 'TRIPD'
363 
364  ! local variables
365  integer npxy, npel, nptot, itmp, jtmp, ktmp, eltmp
366  integer il, jl
367  real xl, yl, xr, yr, rota, rtmp, epsl
368  parameter(epsl = 1.0e-10)
369 
370  real lcoord(LX1*LY1*LZ1*LELT)
371  common /ctmp0/ lcoord
372  integer lmap(LX1*LY1*LZ1*LELT)
373  common /ctmp1/ lmap
374 !-----------------------------------------------------------------------
375  npxy = nx1*ny1
376  npel = npxy*nz1
377  nptot = npel*nelv
378 
379  ! for each line
380  do il=1,trip_nline
381  ! Get coordinates and sort them
382  call copy(lcoord,zm1,nptot)
383  call sort(lcoord,lmap,nptot)
384 
385  ! find unique entrances and provide mapping
386  trip_npoint(il) = 1
387  trip_prj(trip_npoint(il),il) = lcoord(1)
388  itmp = lmap(1)-1
389  eltmp = itmp/npel + 1
390  itmp = itmp - npel*(eltmp-1)
391  ktmp = itmp/npxy + 1
392  itmp = itmp - npxy*(ktmp-1)
393  jtmp = itmp/nx1 + 1
394  itmp = itmp - nx1*(jtmp-1) + 1
395  trip_map(itmp,jtmp,ktmp,eltmp,il) = trip_npoint(il)
396  do jl=2,nptot
397  if((lcoord(jl)-trip_prj(trip_npoint(il),il)).gt.
398  $ max(epsl,abs(epsl*lcoord(jl)))) then
399  trip_npoint(il) = trip_npoint(il) + 1
400  trip_prj(trip_npoint(il),il) = lcoord(jl)
401  endif
402 
403  itmp = lmap(jl)-1
404  eltmp = itmp/npel + 1
405  itmp = itmp - npel*(eltmp-1)
406  ktmp = itmp/npxy + 1
407  itmp = itmp - npxy*(ktmp-1)
408  jtmp = itmp/nx1 + 1
409  itmp = itmp - nx1*(jtmp-1) + 1
410  trip_map(itmp,jtmp,ktmp,eltmp,il) = trip_npoint(il)
411  enddo
412 
413  ! rescale 1D array
414  do jl=1,trip_npoint(il)
415  trip_prj(jl,il) = (trip_prj(jl,il) - trip_spos(ldim,il))
416  $ *trip_ilngt(il)
417  enddo
418 
419  ! get smoothing profile
420  rota = trip_rota(il)
421 
422  do jl=1,nptot
423  itmp = jl-1
424  eltmp = itmp/npel + 1
425  itmp = itmp - npel*(eltmp-1)
426  ktmp = itmp/npxy + 1
427  itmp = itmp - npxy*(ktmp-1)
428  jtmp = itmp/nx1 + 1
429  itmp = itmp - nx1*(jtmp-1) + 1
430 
431  ! rotation
432  xl = xm1(itmp,jtmp,ktmp,eltmp)-trip_spos(1,il)
433  yl = ym1(itmp,jtmp,ktmp,eltmp)-trip_spos(2,il)
434 
435  xr = xl*cos(rota)+yl*sin(rota)
436  yr = -xl*sin(rota)+yl*cos(rota)
437 
438  rtmp = (xr*trip_ismth(1,il))**2 + (yr*trip_ismth(2,il))**2
439  ! Gauss
440  !trip_fsmth(itmp,jtmp,ktmp,eltmp,il) = exp(-4.0*rtmp)
441  ! limited support
442  if (rtmp.lt.1.0) then
443  trip_fsmth(itmp,jtmp,ktmp,eltmp,il) =
444  $ exp(-rtmp)*(1-rtmp)**2
445  else
446  trip_fsmth(itmp,jtmp,ktmp,eltmp,il) = 0.0
447  endif
448 
449  enddo
450  enddo
451 
452  return
453  end subroutine

577  subroutine trip_frcs_get(ifreset)
578  implicit none
579 
580  include 'SIZE'
581  include 'INPUT'
582  include 'TSTEP'
583  include 'TRIPD'
584 
585  ! argument list
586  logical ifreset
587 
588 #ifdef TRIP_PR_RST
589  ! variables necessary to reset pressure projection for P_n-P_n-2
590  integer nprv(2)
591  common /orthbi/ nprv
592 
593  ! variables necessary to reset velocity projection for P_n-P_n-2
594  include 'VPROJ'
595 #endif
596  ! local variables
597  integer il, jl, kl, ll
598  integer istart
599  real theta0, theta
600 
601 #ifdef DEBUG
602  character*3 str1, str2
603  integer iunit, ierr
604  ! call number
605  integer icalldl
606  save icalldl
607  data icalldl /0/
608 #endif
609 !-----------------------------------------------------------------------
610  ! reset all
611  if (ifreset) then
612  if (trip_tiamp.gt.0.0) then
613  istart = 1
614  else
615  istart = 2
616  endif
617  do il= 1, trip_nline
618  do jl = istart, trip_nset_max
619  call rzero(trip_frcs(1,jl,il),trip_npoint(il))
620  do kl= 1, trip_npoint(il)
621  theta0 = 2*pi*trip_prj(kl,il)
622  do ll= 1, trip_nmode(il)
623  theta = theta0*ll
624  trip_frcs(kl,jl,il) = trip_frcs(kl,jl,il) +
625  $ sin(theta+trip_rphs(ll,jl,il))
626  enddo
627  enddo
628  enddo
629  enddo
630  ! rescale time independent part
631  if (trip_tiamp.gt.0.0) then
632  do il= 1, trip_nline
633  call cmult(trip_frcs(1,1,il),trip_tiamp,trip_npoint(il))
634  enddo
635  endif
636  else
637  ! reset only time dependent part if needed
638  if (trip_ntdt.ne.trip_ntdt_old) then
639 #ifdef TRIP_PR_RST
640  ! reset projection space
641  ! pressure
642  if (int(param(95)).gt.0) then
643  param(95) = istep
644  nprv(1) = 0 ! veloctiy field only
645  endif
646  ! velocity
647  if (int(param(94)).gt.0) then
648  param(94) = istep!+2
649  ivproj(2,1) = 0
650  ivproj(2,2) = 0
651  if (if3d) ivproj(2,3) = 0
652  endif
653 #endif
654  do il= 1, trip_nline
655  do jl= trip_nset_max,3,-1
656  call copy(trip_frcs(1,jl,il),trip_frcs(1,jl-1,il),
657  $ trip_npoint(il))
658  enddo
659  call rzero(trip_frcs(1,2,il),trip_npoint(il))
660  do jl= 1, trip_npoint(il)
661  theta0 = 2*pi*trip_prj(jl,il)
662  do kl= 1, trip_nmode(il)
663  theta = theta0*kl
664  trip_frcs(jl,2,il) = trip_frcs(jl,2,il) +
665  $ sin(theta+trip_rphs(kl,2,il))
666  enddo
667  enddo
668  enddo
669  endif
670  endif
671 
672  ! get tripping for current time step
673  if (trip_tiamp.gt.0.0) then
674  do il= 1, trip_nline
675  call copy(trip_ftrp(1,il),trip_frcs(1,1,il),trip_npoint(il))
676  enddo
677  else
678  do il= 1, trip_nline
679  call rzero(trip_ftrp(1,il),trip_npoint(il))
680  enddo
681  endif
682  ! interpolation in time
683  do il = 1, trip_nline
684  theta0= time/trip_tdt(il)-real(trip_ntdt)
685  if (theta0.gt.0.0) then
686  theta0=theta0*theta0*(3.0-2.0*theta0)
687  !theta0=theta0*theta0*theta0*(10.0+(6.0*theta0-15.0)*theta0)
688  do jl= 1, trip_npoint(il)
689  trip_ftrp(jl,il) = trip_ftrp(jl,il) +
690  $ trip_tdamp*((1.0-theta0)*trip_frcs(jl,3,il) +
691  $ theta0*trip_frcs(jl,2,il))
692  enddo
693  else
694  theta0=theta0+1.0
695  theta0=theta0*theta0*(3.0-2.0*theta0)
696  !theta0=theta0*theta0*theta0*(10.0+(6.0*theta0-15.0)*theta0)
697  do jl= 1, trip_npoint(il)
698  trip_ftrp(jl,il) = trip_ftrp(jl,il) +
699  $ trip_tdamp*((1.0-theta0)*trip_frcs(jl,4,il) +
700  $ theta0*trip_frcs(jl,3,il))
701  enddo
702  endif
703  enddo
704 
705 #ifdef DEBUG
706  ! for testing
