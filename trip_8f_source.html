355  subroutine trip_1dprj()
356  implicit none
357 
358  include 'SIZE'
359  include 'INPUT'
360  include 'GEOM'
361  include 'TRIPD'
362 
363  ! local variables
364  integer npxy, npel, nptot, itmp, jtmp, ktmp, eltmp
365  integer il, jl
366  real xl, yl, xr, yr, rota, rtmp, epsl
367  parameter(epsl = 1.0e-10)
368 
369  real lcoord(lx1*ly1*lz1*lelt)
370  common /ctmp0/ lcoord
371  integer lmap(lx1*ly1*lz1*lelt)
372  common /ctmp1/ lmap
373 !-----------------------------------------------------------------------
374  npxy = nx1*ny1
375  npel = npxy*nz1
376  nptot = npel*nelv
377 
378  ! for each line
379  do il=1,trip_nline
380  ! Get coordinates and sort them
381  call copy(lcoord,zm1,nptot)
382  call sort(lcoord,lmap,nptot)
383 
384  ! find unique entrances and provide mapping
385  trip_npoint(il) = 1
386  trip_prj(trip_npoint(il),il) = lcoord(1)
387  itmp = lmap(1)-1
388  eltmp = itmp/npel + 1
389  itmp = itmp - npel*(eltmp-1)
390  ktmp = itmp/npxy + 1
391  itmp = itmp - npxy*(ktmp-1)
392  jtmp = itmp/nx1 + 1
393  itmp = itmp - nx1*(jtmp-1) + 1
394  trip_map(itmp,jtmp,ktmp,eltmp,il) = trip_npoint(il)
395  do jl=2,nptot
396  if((lcoord(jl)-trip_prj(trip_npoint(il),il)).gt.
397  $ max(epsl,abs(epsl*lcoord(jl)))) then
398  trip_npoint(il) = trip_npoint(il) + 1
399  trip_prj(trip_npoint(il),il) = lcoord(jl)
400  endif
401 
402  itmp = lmap(jl)-1
403  eltmp = itmp/npel + 1
404  itmp = itmp - npel*(eltmp-1)
405  ktmp = itmp/npxy + 1
406  itmp = itmp - npxy*(ktmp-1)
407  jtmp = itmp/nx1 + 1
408  itmp = itmp - nx1*(jtmp-1) + 1
409  trip_map(itmp,jtmp,ktmp,eltmp,il) = trip_npoint(il)
410  enddo
411 
412  ! rescale 1D array
413  do jl=1,trip_npoint(il)
414  trip_prj(jl,il) = (trip_prj(jl,il) - trip_spos(ldim,il))
415  $ *trip_ilngt(il)
416  enddo
417 
418  ! get smoothing profile
419  rota = trip_rota(il)
420 
421  do jl=1,nptot
422  itmp = jl-1
423  eltmp = itmp/npel + 1
424  itmp = itmp - npel*(eltmp-1)
425  ktmp = itmp/npxy + 1
426  itmp = itmp - npxy*(ktmp-1)
427  jtmp = itmp/nx1 + 1
428  itmp = itmp - nx1*(jtmp-1) + 1
429 
430  ! rotation
431  xl = xm1(itmp,jtmp,ktmp,eltmp)-trip_spos(1,il)
432  yl = ym1(itmp,jtmp,ktmp,eltmp)-trip_spos(2,il)
433 
434  xr = xl*cos(rota)+yl*sin(rota)
435  yr = -xl*sin(rota)+yl*cos(rota)
436 
437  rtmp = (xr*trip_ismth(1,il))**2 + (yr*trip_ismth(2,il))**2
438  ! Gauss
439  !trip_fsmth(itmp,jtmp,ktmp,eltmp,il) = exp(-4.0*rtmp)
440  ! limited support
441  if (rtmp.lt.1.0) then
442  trip_fsmth(itmp,jtmp,ktmp,eltmp,il) =
443  $ exp(-rtmp)*(1-rtmp)**2
444  else
445  trip_fsmth(itmp,jtmp,ktmp,eltmp,il) = 0.0
446  endif
447 
448  enddo
449  enddo
450 
451  return
452  end subroutine

576  subroutine trip_frcs_get(ifreset)
577  implicit none
578 
579  include 'SIZE'
580  include 'INPUT'
581  include 'TSTEP'
582  include 'TRIPD'
583 
584  ! argument list
585  logical ifreset
586 
587 #ifdef TRIP_PR_RST
588  ! variables necessary to reset pressure projection for P_n-P_n-2
589  integer nprv(2)
590  common /orthbi/ nprv
591 
592  ! variables necessary to reset velocity projection for P_n-P_n-2
593  include 'VPROJ'
594 #endif
595  ! local variables
596  integer il, jl, kl, ll
597  integer istart
598  real theta0, theta
599 
600 #ifdef DEBUG
601  character*3 str1, str2
602  integer iunit, ierr
603  ! call number
604  integer icalldl
605  save icalldl
606  data icalldl /0/
607 #endif
608 !-----------------------------------------------------------------------
609  ! reset all
610  if (ifreset) then
611  if (trip_tiamp.gt.0.0) then
612  istart = 1
613  else
614  istart = 2
615  endif
616  do il= 1, trip_nline
617  do jl = istart, trip_nset_max
618  call rzero(trip_frcs(1,jl,il),trip_npoint(il))
619  do kl= 1, trip_npoint(il)
620  theta0 = 2*pi*trip_prj(kl,il)
621  do ll= 1, trip_nmode(il)
622  theta = theta0*ll
623  trip_frcs(kl,jl,il) = trip_frcs(kl,jl,il) +
624  $ sin(theta+trip_rphs(ll,jl,il))
625  enddo
626  enddo
627  enddo
628  enddo
629  ! rescale time independent part
630  if (trip_tiamp.gt.0.0) then
631  do il= 1, trip_nline
632  call cmult(trip_frcs(1,1,il),trip_tiamp,trip_npoint(il))
633  enddo
634  endif
635  else
636  ! reset only time dependent part if needed
637  if (trip_ntdt.ne.trip_ntdt_old) then
638 #ifdef TRIP_PR_RST
639  ! reset projection space
640  ! pressure
641  if (int(param(95)).gt.0) then
642  param(95) = istep
643  nprv(1) = 0 ! veloctiy field only
644  endif
645  ! velocity
646  if (int(param(94)).gt.0) then
647  param(94) = istep!+2
648  ivproj(2,1) = 0
649  ivproj(2,2) = 0
650  if (if3d) ivproj(2,3) = 0
651  endif
652 #endif
653  do il= 1, trip_nline
654  do jl= trip_nset_max,3,-1
655  call copy(trip_frcs(1,jl,il),trip_frcs(1,jl-1,il),
656  $ trip_npoint(il))
657  enddo
658  call rzero(trip_frcs(1,2,il),trip_npoint(il))
659  do jl= 1, trip_npoint(il)
660  theta0 = 2*pi*trip_prj(jl,il)
661  do kl= 1, trip_nmode(il)
662  theta = theta0*kl
663  trip_frcs(jl,2,il) = trip_frcs(jl,2,il) +
664  $ sin(theta+trip_rphs(kl,2,il))
665  enddo
666  enddo
667  enddo
668  endif
669  endif
670 
671  ! get tripping for current time step
672  if (trip_tiamp.gt.0.0) then
673  do il= 1, trip_nline
674  call copy(trip_ftrp(1,il),trip_frcs(1,1,il),trip_npoint(il))
675  enddo
676  else
677  do il= 1, trip_nline
678  call rzero(trip_ftrp(1,il),trip_npoint(il))
679  enddo
680  endif
681  ! interpolation in time
682  do il = 1, trip_nline
683  theta0= time/trip_tdt(il)-real(trip_ntdt)
684  if (theta0.gt.0.0) then
685  theta0=theta0*theta0*(3.0-2.0*theta0)
686  !theta0=theta0*theta0*theta0*(10.0+(6.0*theta0-15.0)*theta0)
687  do jl= 1, trip_npoint(il)
688  trip_ftrp(jl,il) = trip_ftrp(jl,il) +
689  $ trip_tdamp*((1.0-theta0)*trip_frcs(jl,3,il) +
690  $ theta0*trip_frcs(jl,2,il))
691  enddo
692  else
693  theta0=theta0+1.0
694  theta0=theta0*theta0*(3.0-2.0*theta0)
695  !theta0=theta0*theta0*theta0*(10.0+(6.0*theta0-15.0)*theta0)
696  do jl= 1, trip_npoint(il)
697  trip_ftrp(jl,il) = trip_ftrp(jl,il) +
698  $ trip_tdamp*((1.0-theta0)*trip_frcs(jl,4,il) +
699  $ theta0*trip_frcs(jl,3,il))
700  enddo
701  endif
702  enddo
703 
704 #ifdef DEBUG
705  ! for testing
