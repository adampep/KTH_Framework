728  subroutine get_multi_seg(nelxyz,x,y,z,m,if3d)
729 c
730  integer nelxyz(3)
731  real x(0:m),y(0:m),z(0:m)
732  logical if3d
733 c
734  integer nels (1000)
735  real gains(1000)
736  real xs (0:1000)
737 c
738 c This routine allows the user to input a complex sequence of
739 c segments for each of the x, y and z directions, so that
740 c genbox will generate the desired tensor product array.
741 c
742 c Input format for each coordinat is
743 c
744 c nsegs
745 c nel_1 nel_2 ... nel_nsegs
746 c gain_1 gain_2 ... gain_nsegs
747 c x_0, x_1, x_2, ... x_nsegs
748 c
749 c The output will be:
750 c
751 c x(0)...x(nel_1)...x_(nel_2)...x(nel_segs),
752 c
753 c where the segment between x(e-1) and x(e) is filled with
754 c a distribution determined by gain_e. Uniform spacing
755 c corresponds to gain_e = 1, otherwise, a geometric sequence
756 c is generated, with dx_i+1 = dx_i * gain_e
757 c
758  ndim = 2
759  if (if3d) ndim=3
760 c
761  do id=1,ndim
762 c
763  call geti1(nseg,iend,7)
764  call getiv(nels ,nseg ,iend,7)
765  call getrv(xs ,nseg+1,iend,7)
766  call getrv(gains,nseg ,iend,7)
767  write(6,*) 'NSEG: ',id,nseg,(nels(k),k=1,nseg)
768 c
769  nelxyz(id) = ilsum(nels,nseg)
770  if (nelxyz(id).gt.m) then
771  write(6,*) 'NEL too large:',nelxyz(id),m,id
772  write(6,*) 'Abort in get_multi_seg.'
773  call exitt
774  endif
775 c
776  k = 0
777  do is=1,nseg
778  nel = nels(is)
779  x0 = xs(is-1)
780  x1 = xs(is)
781  if (id.eq.1) call geometric_x(x(k),nel,x0,x1,gains(is))
782  if (id.eq.2) call geometric_x(y(k),nel,x0,x1,gains(is))
783  if (id.eq.3) call geometric_x(z(k),nel,x0,x1,gains(is))
784  k = k + nel
785  enddo
786  enddo
787 c
788  return
789  end
790 c-----------------------------------------------------------------------
791  subroutine geometric_x(x,n,x0,x1,gain)
792 c
793  real x(0:n)
794 c
795  x(0) = 0.
796  dx = (x1-x0)/n
797  do i=1,n
798  x(i) = x(i-1) + dx
799  dx = dx*gain
800  enddo
801 c
802  scale = (x1-x0)/(x(n)-x(0))
803  do i=0,n
804  x(i) = x0 + scale*x(i)
805  enddo
806 c
807  return
808  end

811  real x(0:2)

813 c
814  x0 = x(0)
815  x1 = x(1)

817  write(6,*) 'x0:',x0,x1,r,nelx
818 c
